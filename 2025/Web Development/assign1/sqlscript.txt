-- Create status table
CREATE TABLE IF NOT EXISTS status (
                                    stcode VARCHAR(5) PRIMARY KEY,
                                    status TEXT NOT NULL,
                                    share VARCHAR(20),
                                    date DATE NOT NULL,
                                    permission TEXT
                                    );
